class ScreenAdapter {
  static designWidth = 1920;
  static designHeight = 1080;
  
  static getScale() {
    const width = window.innerWidth;
    const height = window.innerHeight;
    
    return {
      widthScale: width / this.designWidth,
      heightScale: height / this.designHeight,
      minScale: Math.min(width / this.designWidth, height / this.designHeight)
    };
  }
  
  static isSmallScreen() {
    return window.innerWidth < 1280;
  }
  
  static isLargeScreen() {
    return window.innerWidth > 1920;
  }
  
  static getOptimalFontSize(baseSize) {
    const { widthScale } = this.getScale();
    return Math.max(baseSize * 0.8, baseSize * widthScale);
  }
}

// React Hookç‰ˆæœ¬
function useScreenAdapter() {
  const [screenInfo, setScreenInfo] = useState({
    isSmall: ScreenAdapter.isSmallScreen(),
    isLarge: ScreenAdapter.isLargeScreen(),
    scale: ScreenAdapter.getScale()
  });
  
  useEffect(() => {
    const handleResize = () => {
      setScreenInfo({
        isSmall: ScreenAdapter.isSmallScreen(),
        isLarge: ScreenAdapter.isLargeScreen(),
        scale: ScreenAdapter.getScale()
      });
    };
    
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);
  
  return screenInfo;
}
