使用 `fetch` + `Response.body.getReader()` 实现下载进度监听，这是原生 fetch 唯一支持进度反馈的方式。

## 完整实现代码

```javascript
async function downloadWithProgress(url, options = {}) {
    const {
        onProgress,      // 进度回调 (loaded, total, percent)
        headers = {},    // 自定义请求头
        method = 'GET'
    } = options;

    try {
        // 发起请求
        const response = await fetch(url, {
            method,
            headers,
            credentials: 'include'
        });

        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }

        // 获取文件总大小（字节）
        const contentLength = response.headers.get('Content-Length');
        const total = contentLength ? parseInt(contentLength, 10) : null;
        
        // 获取 reader
        const reader = response.body.getReader();
        
        // 收集数据块
        const chunks = [];
        let receivedLength = 0;
        
        // 读取流
        while (true) {
            const { done, value } = await reader.read();
            
            if (done) break;
            
            // 存储数据块
            chunks.push(value);
            receivedLength += value.length;
            
            // 触发进度回调
            if (onProgress && total) {
                const percent = (receivedLength / total) * 100;
                onProgress({
                    loaded: receivedLength,
                    total: total,
                    percent: Math.round(percent)
                });
            } else if (onProgress) {
                onProgress({
                    loaded: receivedLength,
                    total: null,
                    percent: null
                });
            }
        }
        
        // 合并所有数据块
        const blob = new Blob(chunks);
        
        // 触发完成回调
        if (onProgress) {
            onProgress({
                loaded: blob.size,
                total: blob.size,
                percent: 100,
                complete: true
            });
        }
        
        return blob;
        
    } catch (error) {
        console.error('下载失败:', error);
        throw error;
    }
}
```

```javascript
// 辅助函数：格式化字节数
function formatBytes(bytes) {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
}

// 保存 Blob 为文件
function saveBlobAsFile(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
}
```

### 2. 边下载边保存到磁盘（内存最优方案）

配合 File System Access API，避免大文件内存溢出：

```javascript
async function downloadToDisk(url, suggestedName, options = {}) {
    const { headers = {}, onProgress } = options;
    
    try {
        // 请求用户选择保存位置
        const fileHandle = await window.showSaveFilePicker({
            suggestedName,
            types: [{
                description: '下载文件',
                accept: { 'application/octet-stream': ['.*'] }
            }]
        });
        
        const writable = await fileHandle.createWritable();
        
        const response = await fetch(url, { headers });
        const contentLength = response.headers.get('Content-Length');
        const total = contentLength ? parseInt(contentLength, 10) : null;
        
        const reader = response.body.getReader();
        let receivedLength = 0;
        
        while (true) {
            const { done, value } = await reader.read();
            if (done) break;
            
            // 直接写入磁盘，不占用内存
            await writable.write(value);
            receivedLength += value.length;
            
            if (onProgress && total) {
                const percent = (receivedLength / total) * 100;
                onProgress({
                    loaded: receivedLength,
                    total: total,
                    percent: Math.round(percent)
                });
            }
        }
        
        await writable.close();
        
        return { success: true, path: fileHandle.name };
        
    } catch (error) {
        if (error.name !== 'AbortError') {
            console.error('下载失败:', error);
        }
        throw error;
    }
}

        alert('下载失败: ' + error.message);
```

### 3. 支持暂停/恢复的下载器

```javascript
class DownloadManager {
    constructor() {
        this.abortController = null;
        this.isPaused = false;
        this.resumeData = null;
    }
    
    async download(url, options = {}) {
        const { headers = {}, onProgress } = options;
        
        this.abortController = new AbortController();
        this.isPaused = false;
        
        try {
            // 支持断点续传
            const requestHeaders = { ...headers };
            if (this.resumeData) {
                requestHeaders['Range'] = `bytes=${this.resumeData.loaded}-`;
            }
            
            const response = await fetch(url, {
                headers: requestHeaders,
                signal: this.abortController.signal
            });
            
            const total = parseInt(response.headers.get('Content-Length'), 10) + 
                         (this.resumeData?.loaded || 0);
            const reader = response.body.getReader();
            const chunks = [];
            let receivedLength = this.resumeData?.loaded || 0;
            
            while (true) {
                // 检查是否暂停
                if (this.isPaused) {
                    this.resumeData = { loaded: receivedLength };
                    return { paused: true, loaded: receivedLength };
                }
                
                const { done, value } = await reader.read();
                if (done) break;
                
                chunks.push(value);
                receivedLength += value.length;
                
                if (onProgress) {
                    const percent = (receivedLength / total) * 100;
                    onProgress({
                        loaded: receivedLength,
                        total: total,
                        percent: Math.round(percent)
                    });
                }
            }
            
            // 下载完成，清除恢复数据
            this.resumeData = null;
            
            // 合并数据
            const blob = new Blob(chunks);
            return { blob, loaded: receivedLength, total };
            
        } catch (error) {
            if (error.name === 'AbortError') {
                console.log('下载已取消');
            }
            throw error;
        }
    }
    
    pause() {
        this.isPaused = true;
        if (this.abortController) {
            this.abortController.abort();
        }
    }
    
    resume() {
        return this.download(); // 重新调用，会自动从断点继续
    }
    
    cancel() {
        this.isPaused = false;
        this.resumeData = null;
        if (this.abortController) {
            this.abortController.abort();
        }
    }
}

// 使用示例
const manager = new DownloadManager();

// 开始下载
manager.download('https://example.com/big-file.zip', {
    headers: { 'Authorization': 'Bearer token' },
    onProgress: (progress) => {
        console.log(`${progress.percent}%`);
    }
});

// 暂停
setTimeout(() => manager.pause(), 5000);

// 恢复
setTimeout(() => manager.resume(), 10000);
```

## 核心要点总结

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **fetch + reader** | 原生支持，可控性强 | 代码稍复杂 | 需要进度反馈的场景 |
| **axios + onDownloadProgress** | 代码简单，功能完善 | 需引入第三方库 | 项目已使用 axios |
| **边下载边写磁盘** | 内存占用极小 | 需要 File System API | 超大文件（>500MB） |

## 注意事项

1. **Content-Length 可能不存在**：某些服务器不返回这个头，此时无法计算百分比，只能显示已下载大小
2. **CORS 问题**：跨域请求需要服务器支持，且需要暴露 `Content-Length` 头
3. **内存占用**：使用 `chunks.push(value)` 会在内存中累积所有数据，超大文件建议使用 File System API 边下载边写入磁盘
4. **错误处理**：网络中断、服务器错误等情况需要妥善处理

如果你的文件不大（<100MB），使用第一种方案就足够了。如果需要下载几百 MB 甚至 GB 级别的文件，强烈建议使用第二种方案（边下载边写磁盘）。
