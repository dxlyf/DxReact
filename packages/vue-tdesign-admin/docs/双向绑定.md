## 📝 Vue 3 通过 model 传入数组并实现双向绑定

以下是完整的实现方案，包括父组件传递数组、子组件循环渲染并实现双向绑定：

## 🎯 方案一：使用 v-model 绑定数组（推荐）

### 1. **子组件 - ListEditor.vue**

```vue
<template>
  <div class="list-editor">
    <div class="list-header">
      <h3>列表编辑器 (共 {{ modelValue.length }} 项)</h3>
      <button @click="addItem" class="add-btn">➕ 新增</button>
    </div>

    <div class="list-items">
      <div
        v-for="(item, index) in modelValue"
        :key="index"
        class="list-item"
      >
        <div class="item-index">{{ index + 1 }}.</div>
        
        <!-- 使用 v-model 直接绑定数组元素 -->
        <input
          v-model="modelValue[index]"
          type="text"
          :placeholder="`输入第 ${index + 1} 项`"
          class="item-input"
          @input="handleInput(index, $event)"
        />
        
        <div class="item-actions">
          <button @click="moveUp(index)" :disabled="index === 0" class="icon-btn">↑</button>
          <button @click="moveDown(index)" :disabled="index === modelValue.length - 1" class="icon-btn">↓</button>
          <button @click="removeItem(index)" class="icon-btn delete">🗑️</button>
        </div>
      </div>
    </div>

    <!-- 空状态提示 -->
    <div v-if="modelValue.length === 0" class="empty-state">
      暂无数据，点击"新增"按钮添加
    </div>

    <!-- 调试信息（可选） -->
    <div class="debug-info" v-if="showDebug">
      <pre>当前数据: {{ JSON.stringify(modelValue, null, 2) }}</pre>
    </div>
  </div>
</template>

<script setup>
import { ref, watch } from 'vue'

// 定义 props，接收数组
const props = defineProps({
  modelValue: {
    type: Array,
    required: true,
    default: () => []
  },
  showDebug: {
    type: Boolean,
    default: false
  }
})

// 定义事件
const emit = defineEmits(['update:modelValue'])

/**
 * 处理输入变化
 * 方式1：直接通过 v-model 已经绑定了，这个函数可以用于额外处理
 */
const handleInput = (index, event) => {
  // v-model 已经自动更新了 modelValue[index]
  // 这里可以添加额外逻辑，如验证、格式化等
  console.log(`第 ${index + 1} 项修改为:`, event.target.value)
}

/**
 * 添加新项
 */
const addItem = () => {
  const newList = [...props.modelValue, '']
  emit('update:modelValue', newList)
}

/**
 * 删除项
 */
const removeItem = (index) => {
  const newList = props.modelValue.filter((_, i) => i !== index)
  emit('update:modelValue', newList)
}

/**
 * 上移
 */
const moveUp = (index) => {
  if (index === 0) return
  const newList = [...props.modelValue]
  ;[newList[index - 1], newList[index]] = [newList[index], newList[index - 1]]
  emit('update:modelValue', newList)
}

/**
 * 下移
 */
const moveDown = (index) => {
  if (index === props.modelValue.length - 1) return
  const newList = [...props.modelValue]
  ;[newList[index], newList[index + 1]] = [newList[index + 1], newList[index]]
  emit('update:modelValue', newList)
}

/**
 * 批量操作示例
 */
const batchUpdate = (updater) => {
  const newList = props.modelValue.map(updater)
  emit('update:modelValue', newList)
}

// 暴露方法给父组件
defineExpose({
  addItem,
  removeItem,
  batchUpdate
})
</script>
```

### 2. **父组件 - ParentComponent.vue**

```vue
<template>
  <div class="parent-container">
    <h2>数组编辑器示例</h2>
    
    <!-- 基础用法：直接绑定数组 -->
    <ListEditor v-model="items" showDebug />
    
    <!-- 带默认值的用法 -->
    <ListEditor v-model="presetItems" />
    
    <!-- 通过 ref 调用子组件方法 -->
    <ListEditor 
      ref="editorRef"
      v-model="dynamicItems"
    />
    
    <div class="actions">
      <button @click="handleBatchUpdate">批量更新</button>
      <button @click="handleReset">重置</button>
      <button @click="logCurrentData">查看数据</button>
    </div>
    
    <div class="preview">
      <h4>当前数据预览:</h4>
      <pre>{{ JSON.stringify(items, null, 2) }}</pre>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import ListEditor from './ListEditor.vue'

// 示例1：基础数据
const items = ref(['苹果', '香蕉', '橙子'])

// 示例2：预设数据
const presetItems = ref(['Vue', 'React', 'Angular'])

// 示例3：动态数据
const dynamicItems = ref(['Item 1', 'Item 2', 'Item 3'])

// 获取子组件引用
const editorRef = ref(null)

// 批量更新示例
const handleBatchUpdate = () => {
  // 通过 v-model 直接更新
  items.value = items.value.map(item => item + ' (已更新)')
  
  // 或者调用子组件的方法
  if (editorRef.value) {
    editorRef.value.batchUpdate(item => item.toUpperCase())
  }
}

// 重置数据
const handleReset = () => {
  items.value = ['苹果', '香蕉', '橙子']
}

// 查看当前数据
const logCurrentData = () => {
  console.log('当前数据:', items.value)
}
</script>
```

## 🔄 方案二：使用 computed 实现更精细的控制

```vue
<template>
  <div class="controlled-list-editor">
    <div
      v-for="(item, index) in localList"
      :key="index"
      class="list-item"
    >
      <input
        :value="item"
        @input="updateItem(index, $event.target.value)"
        type="text"
        class="item-input"
      />
      <button @click="removeItem(index)" class="remove-btn">删除</button>
    </div>
    
    <button @click="addItem" class="add-btn">添加</button>
  </div>
</template>

<script setup>
import { computed } from 'vue'

const props = defineProps({
  modelValue: {
    type: Array,
    required: true
  }
})

const emit = defineEmits(['update:modelValue'])

// 使用 computed 创建本地响应式副本
const localList = computed({
  get: () => props.modelValue,
  set: (value) => emit('update:modelValue', value)
})

// 更新单个项
const updateItem = (index, value) => {
  const newList = [...props.modelValue]
  newList[index] = value
  emit('update:modelValue', newList)
}

// 添加项
const addItem = () => {
  emit('update:modelValue', [...props.modelValue, ''])
}

// 删除项
const removeItem = (index) => {
  const newList = props.modelValue.filter((_, i) => i !== index)
  emit('update:modelValue', newList)
}
</script>
```

## 🎨 方案三：使用 defineModel (Vue 3.4+)

```vue
<template>
  <div class="modern-list-editor">
    <div v-for="(item, index) in model" :key="index" class="item-row">
      <input v-model="model[index]" type="text" class="item-input" />
      <button @click="removeItem(index)" class="remove-btn">×</button>
    </div>
    <button @click="addItem" class="add-btn">+ 添加</button>
  </div>
</template>

<script setup>
// Vue 3.4+ 使用 defineModel
const model = defineModel({
  type: Array,
  default: () => []
})

// 添加项
const addItem = () => {
  model.value = [...model.value, '']
}

// 删除项
const removeItem = (index) => {
  model.value = model.value.filter((_, i) => i !== index)
}

// 可以直接修改数组元素
// v-model="model[index]" 会自动更新数组
</script>
```

## 📊 方案四：带验证和复杂操作的增强版

```vue
<template>
  <div class="advanced-list-editor">
    <div class="toolbar">
      <button @click="addItem" class="btn-primary">➕ 新增</button>
      <button @click="sortAsc" class="btn-secondary">↑ 升序</button>
      <button @click="sortDesc" class="btn-secondary">↓ 降序</button>
      <button @click="removeDuplicates" class="btn-warning">🔄 去重</button>
      <button @click="clearAll" class="btn-danger">🗑️ 清空</button>
    </div>

    <div class="items-container">
      <transition-group name="list" tag="div">
        <div
          v-for="(item, index) in modelValue"
          :key="`${index}-${item}`"
          class="item-card"
        >
          <div class="item-header">
            <span class="item-index">#{{ index + 1 }}</span>
            <span class="item-valid" :class="{ invalid: !isValid(item) }">
              {{ isValid(item) ? '✓ 有效' : '✗ 无效' }}
            </span>
          </div>
          
          <input
            :value="item"
            @input="updateItem(index, $event.target.value)"
            @blur="validateItem(index)"
            type="text"
            class="item-input"
            :class="{ 'input-error': errors[index] }"
          />
          
          <div v-if="errors[index]" class="error-message">
            {{ errors[index] }}
          </div>
          
          <div class="item-actions">
            <button @click="moveUp(index)" :disabled="index === 0" class="icon-btn">↑</button>
            <button @click="moveDown(index)" :disabled="index === modelValue.length - 1" class="icon-btn">↓</button>
            <button @click="removeItem(index)" class="icon-btn delete">🗑️</button>
          </div>
        </div>
      </transition-group>
    </div>

    <div class="stats">
      <span>总数: {{ modelValue.length }}</span>
      <span>有效: {{ validCount }}</span>
      <span>无效: {{ invalidCount }}</span>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'

const props = defineProps({
  modelValue: {
    type: Array,
    required: true
  },
  validationRules: {
    type: Object,
    default: () => ({
      minLength: 1,
      maxLength: 100,
      pattern: null,
      required: true
    })
  }
})

const emit = defineEmits(['update:modelValue'])

// 错误状态
const errors = ref({})

// 验证单个项
const validateItem = (index) => {
  const value = props.modelValue[index]
  const rules = props.validationRules
  
  if (rules.required && (!value || value.trim() === '')) {
    errors.value[index] = '此项不能为空'
    return false
  }
  
  if (rules.minLength && value.length < rules.minLength) {
    errors.value[index] = `最小长度 ${rules.minLength}`
    return false
  }
  
  if (rules.maxLength && value.length > rules.maxLength) {
    errors.value[index] = `最大长度 ${rules.maxLength}`
    return false
  }
  
  if (rules.pattern && !rules.pattern.test(value)) {
    errors.value[index] = '格式不正确'
    return false
  }
  
  delete errors.value[index]
  return true
}

// 检查是否有效
const isValid = (value) => {
  if (props.validationRules.required && (!value || value.trim() === '')) {
    return false
  }
  return true
}

// 统计
const validCount = computed(() => {
  return props.modelValue.filter(isValid).length
})

const invalidCount = computed(() => {
  return props.modelValue.length - validCount.value
})

// 更新项
const updateItem = (index, value) => {
  const newList = [...props.modelValue]
  newList[index] = value
  emit('update:modelValue', newList)
  
  // 实时验证
  validateItem(index)
}

// 添加项
const addItem = () => {
  const newList = [...props.modelValue, '']
  emit('update:modelValue', newList)
}

// 删除项
const removeItem = (index) => {
  const newList = props.modelValue.filter((_, i) => i !== index)
  emit('update:modelValue', newList)
  
  // 清理错误
  const newErrors = { ...errors.value }
  delete newErrors[index]
  errors.value = newErrors
}

// 上移
const moveUp = (index) => {
  if (index === 0) return
  const newList = [...props.modelValue]
  ;[newList[index - 1], newList[index]] = [newList[index], newList[index - 1]]
  emit('update:modelValue', newList)
}

// 下移
const moveDown = (index) => {
  if (index === props.modelValue.length - 1) return
  const newList = [...props.modelValue]
  ;[newList[index], newList[index + 1]] = [newList[index + 1], newList[index]]
  emit('update:modelValue', newList)
}

// 升序排序
const sortAsc = () => {
  const newList = [...props.modelValue].sort((a, b) => a.localeCompare(b))
  emit('update:modelValue', newList)
}

// 降序排序
const sortDesc = () => {
  const newList = [...props.modelValue].sort((a, b) => b.localeCompare(a))
  emit('update:modelValue', newList)
}

// 去重
const removeDuplicates = () => {
  const newList = [...new Set(props.modelValue)]
  emit('update:modelValue', newList)
}

// 清空
const clearAll = () => {
  emit('update:modelValue', [])
  errors.value = {}
}
</script>
```

## 📝 总结

| 方案 | 适用场景 | 优点 | 缺点 |
|:---|:---|:---|:---|
| 方案一：v-model | 基础场景 | 简单直观 | 功能较少 |
| 方案二：computed | 需要控制 | 灵活可控 | 代码稍多 |
| 方案三：defineModel | Vue 3.4+ | 最简洁 | 版本限制 |
| 方案四：增强版 | 复杂场景 | 功能完善 | 代码复杂 |

### 💡 最佳实践建议

1. **基础需求**：使用方案一（v-model）
2. **Vue 3.4+**：使用方案三（defineModel）
3. **需要验证**：使用方案四
4. **性能优化**：使用 `key` 绑定唯一值，避免使用索引作为 key
5. **大列表**：考虑虚拟滚动或分页
