## 📤 文件上传进度实时监控完整方案

当后端处理文件需要较长时间时，前端需要实时知道进度来提升用户体验。以下是几种实现方案：

## 🎯 方案一：XMLHttpRequest 原生进度事件（最简单）

```javascript
// uploadWithProgress.js

/**
 * 使用 XHR 上传文件并监听进度
 */
function uploadFileWithProgress(file, url, options = {}) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest()
    
    // 创建 FormData
    const formData = new FormData()
    formData.append('file', file)
    
    // 可以添加其他参数
    if (options.data) {
      Object.keys(options.data).forEach(key => {
        formData.append(key, options.data[key])
      })
    }

    // 监听上传进度
    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const percentComplete = (e.loaded / e.total) * 100
        options.onProgress?.({
          type: 'upload',
          loaded: e.loaded,
          total: e.total,
          percent: Math.round(percentComplete)
        })
      }
    })

    // 监听下载进度（接收服务器响应）
    xhr.addEventListener('progress', (e) => {
      if (e.lengthComputable && options.onDownloadProgress) {
        options.onDownloadProgress({
          type: 'download',
          loaded: e.loaded,
          total: e.total,
          percent: Math.round((e.loaded / e.total) * 100)
        })
      }
    })

    // 监听状态变化
    xhr.upload.addEventListener('load', () => {
      options.onProgress?.({
        type: 'upload',
        loaded: file.size,
        total: file.size,
        percent: 100,
        status: 'complete'
      })
    })

    xhr.upload.addEventListener('error', () => {
      reject(new Error('上传失败'))
    })

    xhr.upload.addEventListener('abort', () => {
      reject(new Error('上传已取消'))
    })

    // 监听请求完成
    xhr.addEventListener('load', () => {
      if (xhr.status >= 200 && xhr.status < 300) {
        try {
          const response = JSON.parse(xhr.responseText)
          resolve(response)
        } catch {
          resolve(xhr.responseText)
        }
      } else {
        reject(new Error(`HTTP ${xhr.status}: ${xhr.statusText}`))
      }
    })

    xhr.addEventListener('error', () => {
      reject(new Error('网络错误'))
    })

    xhr.addEventListener('abort', () => {
      reject(new Error('请求已取消'))
    })

    // 打开连接
    xhr.open('POST', url, true)
    
    // 设置请求头
    if (options.headers) {
      Object.keys(options.headers).forEach(key => {
        xhr.setRequestHeader(key, options.headers[key])
      })
    }

    // 设置超时
    if (options.timeout) {
      xhr.timeout = options.timeout
    }

    // 发送请求
    xhr.send(formData)

    // 返回取消函数
    return () => xhr.abort()
  })
}
```

## 🔄 方案二：Fetch API + 进度监控（需要 Response 支持）

```javascript
// fetchWithProgress.js

/**
 * 使用 Fetch API 并监控进度
 * 注意：Fetch API 原生不支持上传进度，只能监控下载进度
 */
async function fetchWithProgress(url, options = {}) {
  const {
    file,
    onUploadProgress,
    onDownloadProgress,
    ...fetchOptions
  } = options

  // 创建 FormData
  const formData = new FormData()
  if (file) {
    formData.append('file', file)
  }

  // 对于上传进度，XHR 仍然是更好的选择
  // 但如果必须用 Fetch，可以这样实现下载进度
  const response = await fetch(url, {
    method: 'POST',
    body: formData,
    ...fetchOptions
  })

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}`)
  }

  // 监控下载进度
  if (onDownloadProgress) {
    const contentLength = response.headers.get('content-length')
    const total = contentLength ? parseInt(contentLength, 10) : null
    
    if (total && response.body) {
      const reader = response.body.getReader()
      const chunks = []
      let receivedLength = 0

      while (true) {
        const { done, value } = await reader.read()
        
        if (done) break
        
        chunks.push(value)
        receivedLength += value.length
        
        onDownloadProgress({
          loaded: receivedLength,
          total,
          percent: Math.round((receivedLength / total) * 100)
        })
      }

      // 合并数据
      const blob = new Blob(chunks)
      return {
        ok: response.ok,
        status: response.status,
        headers: response.headers,
        async json() {
          return JSON.parse(await blob.text())
        },
        async text() {
          return await blob.text()
        },
        async blob() {
          return blob
        }
      }
    }
  }

  return response
}
```

## 📡 方案三：WebSocket 实时推送（适合长时间处理）

```javascript
// websocketUpload.js

class WebSocketUploader {
  constructor(options = {}) {
    this.wsUrl = options.wsUrl || 'ws://localhost:8080/ws'
    this.socket = null
    this.uploadId = null
    this.onProgress = options.onProgress
    this.onComplete = options.onComplete
    this.onError = options.onError
  }

  /**
   * 连接 WebSocket
   */
  connect() {
    return new Promise((resolve, reject) => {
      this.socket = new WebSocket(this.wsUrl)
      
      this.socket.onopen = () => {
        console.log('WebSocket 连接成功')
        resolve()
      }
      
      this.socket.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data)
          this.handleMessage(data)
        } catch (e) {
          console.error('解析消息失败:', e)
        }
      }
      
      this.socket.onerror = (error) => {
        console.error('WebSocket 错误:', error)
        this.onError?.(error)
        reject(error)
      }
      
      this.socket.onclose = () => {
        console.log('WebSocket 连接关闭')
      }
    })
  }

  /**
   * 处理服务器推送的消息
   */
  handleMessage(data) {
    switch (data.type) {
      case 'upload:start':
        this.uploadId = data.uploadId
        break
        
      case 'upload:progress':
        this.onProgress?.({
          type: 'processing',
          loaded: data.processed,
          total: data.total,
          percent: data.percent,
          message: data.message,
          stage: data.stage // 'parsing', 'validating', 'importing'
        })
        break
        
      case 'upload:complete':
        this.onComplete?.(data.result)
        this.disconnect()
        break
        
      case 'upload:error':
        this.onError?.(new Error(data.message))
        this.disconnect()
        break
    }
  }

  /**
   * 上传文件
   */
  async uploadFile(file, extraData = {}) {
    if (!this.socket || this.socket.readyState !== WebSocket.OPEN) {
      await this.connect()
    }

    // 创建 FormData
    const formData = new FormData()
    formData.append('file', file)
    Object.keys(extraData).forEach(key => {
      formData.append(key, extraData[key])
    })

    // 发送文件到 HTTP 接口
    const response = await fetch('/api/upload', {
      method: 'POST',
      body: formData,
      headers: {
        'X-WebSocket-Id': this.uploadId // 关联 WebSocket 连接
      }
    })

    if (!response.ok) {
      throw new Error('上传失败')
    }

    return response.json()
  }

  /**
   * 断开连接
   */
  disconnect() {
    if (this.socket) {
      this.socket.close()
      this.socket = null
    }
  }
}
```

## 🔄 方案四：轮询查询进度（最通用）

```javascript
// pollUpload.js

class PollUploader {
  constructor(options = {}) {
    this.pollInterval = options.pollInterval || 2000 // 2秒轮询一次
    this.maxRetries = options.maxRetries || 60 // 最多轮询2分钟
    this.onProgress = options.onProgress
    this.onComplete = options.onComplete
    this.onError = options.onError
    this.pollTimer = null
    this.retryCount = 0
  }

  /**
   * 上传文件
   */
  async uploadFile(file, url, options = {}) {
    const formData = new FormData()
    formData.append('file', file)
    
    if (options.data) {
      Object.keys(options.data).forEach(key => {
        formData.append(key, options.data[key])
      })
    }

    // 1. 上传文件，获取任务ID
    const response = await fetch(url, {
      method: 'POST',
      body: formData,
      headers: options.headers || {}
    })

    if (!response.ok) {
      throw new Error('上传失败')
    }

    const result = await response.json()
    const taskId = result.taskId || result.id
    
    if (!taskId) {
      throw new Error('服务器未返回任务ID')
    }

    // 2. 开始轮询进度
    this.startPolling(taskId, options.progressUrl || `${url}/progress`)
  }

  /**
   * 开始轮询进度
   */
  startPolling(taskId, progressUrl) {
    this.pollTimer = setInterval(async () => {
      try {
        const response = await fetch(`${progressUrl}/${taskId}`)
        
        if (!response.ok) {
          throw new Error('获取进度失败')
        }
        
        const data = await response.json()
        
        // 触发进度回调
        this.onProgress?.({
          taskId,
          loaded: data.processed || 0,
          total: data.total || 100,
          percent: data.percent || 0,
          status: data.status,
          message: data.message,
          stage: data.stage
        })

        // 检查是否完成
        if (data.status === 'completed' || data.percent >= 100) {
          this.complete(taskId, data)
        } else if (data.status === 'failed') {
          this.error(new Error(data.message || '处理失败'))
        }

        this.retryCount = 0 // 重置重试计数

      } catch (error) {
        this.retryCount++
        
        if (this.retryCount >= this.maxRetries) {
          this.error(new Error('轮询超时'))
        }
      }
    }, this.pollInterval)
  }

  /**
   * 完成处理
   */
  complete(taskId, result) {
    this.stopPolling()
    this.onComplete?.({
      taskId,
      ...result
    })
  }

  /**
   * 错误处理
   */
  error(error) {
    this.stopPolling()
    this.onError?.(error)
  }

  /**
   * 停止轮询
   */
  stopPolling() {
    if (this.pollTimer) {
      clearInterval(this.pollTimer)
      this.pollTimer = null
    }
  }
}
```

```vue

    </div>
    
    <div v-if="uploadError" class="result error">

<script setup>
import { ref, reactive } from 'vue'

// 状态
const fileInput = ref(null)
const selectedFile = ref(null)
const isDragover = ref(false)

// 上传相关
const uploading = ref(false)
const uploadProgress = reactive({
  loaded: 0,
  total: 0,
  percent: 0
})

// 处理相关
const processing = ref(false)
const processingStage = ref('')
const processingProgress = reactive({
  loaded: 0,
  total: 0,
  percent: 0
})
const statusMessage = ref('')

// 结果
const uploadResult = ref(null)
const uploadError = ref(null)

// 速度计算
const uploadSpeed = ref('')
let speedTimer = null
let lastLoaded = 0
let lastTime = Date.now()

// 取消控制器
let abortController = null

/**
 * 触发文件选择
 */
const triggerFileInput = () => {
  fileInput.value.click()
}

/**
 * 处理文件选择
 */
const handleFileSelect = (event) => {
  const file = event.target.files[0]
  if (file) {
    selectedFile.value = file
    resetState()
  }
}

/**
 * 处理拖拽
 */
const handleDrop = (event) => {
  isDragover.value = false
  const file = event.dataTransfer.files[0]
  if (file) {
    selectedFile.value = file
    resetState()
  }
}

/**
 * 重置状态
 */
const resetState = () => {
  uploading.value = false
  processing.value = false
  uploadProgress.loaded = 0
  uploadProgress.percent = 0
  processingProgress.loaded = 0
  processingProgress.percent = 0
  uploadResult.value = null
  uploadError.value = null
  uploadSpeed.value = ''
  clearInterval(speedTimer)
}

/**
 * 计算上传速度
 */
const calculateSpeed = (loaded) => {
  const now = Date.now()
  const timeDiff = now - lastTime
  
  if (timeDiff > 500) {
    const bytesDiff = loaded - lastLoaded
    const speedBps = (bytesDiff / timeDiff) * 1000
    
    if (speedBps > 1024 * 1024) {
      uploadSpeed.value = `${(speedBps / (1024 * 1024)).toFixed(2)} MB/s`
    } else if (speedBps > 1024) {
      uploadSpeed.value = `${(speedBps / 1024).toFixed(2)} KB/s`
    } else {
      uploadSpeed.value = `${speedBps.toFixed(0)} B/s`
    }
    
    lastLoaded = loaded
    lastTime = now
  }
}

/**
 * 开始上传
 */
const startUpload = async () => {
  if (!selectedFile.value) return
  
  resetState()
  uploading.value = true
  uploadError.value = null
  
  // 创建取消控制器
  abortController = new AbortController()
  
  // 创建 FormData
  const formData = new FormData()
  formData.append('file', selectedFile.value)
  
  // 使用 XHR 上传（可以监控进度）
  const xhr = new XMLHttpRequest()
  
  xhr.upload.addEventListener('progress', (e) => {
    if (e.lengthComputable) {
      uploadProgress.loaded = e.loaded
      uploadProgress.total = e.total
      uploadProgress.percent = Math.round((e.loaded / e.total) * 100)
      
      calculateSpeed(e.loaded)
    }
  })
  
  xhr.upload.addEventListener('load', () => {
    uploadProgress.percent = 100
    clearInterval(speedTimer)
  })
  
  xhr.addEventListener('load', () => {
    if (xhr.status >= 200 && xhr.status < 300) {
      try {
        const response = JSON.parse(xhr.responseText)
        
        // 上传完成，开始轮询处理进度
        uploading.value = false
        processing.value = true
        
        // 开始轮询处理进度
        pollProcessingProgress(response.taskId)
      } catch (e) {
        handleError('解析响应失败')
      }
    } else {
      handleError(`上传失败: ${xhr.status}`)
    }
  })
  
  xhr.addEventListener('error', () => {
    handleError('网络错误')
  })
  
  xhr.addEventListener('abort', () => {
    handleError('上传已取消')
  })
  
  // 发送请求
  xhr.open('POST', '/api/upload/large-file')
  
  // 设置请求头
  xhr.setRequestHeader('X-Requested-With', 'XMLHttpRequest')
  
  // 发送
  xhr.send(formData)
  
  // 保存 xhr 用于取消
  window.currentXHR = xhr
}

/**
 * 轮询处理进度
 */
const pollProcessingProgress = async (taskId) => {
  const pollInterval = 2000 // 2秒
  let retryCount = 0
  const maxRetries = 60 // 最多轮询2分钟
  
  const poll = async () => {
    try {
      const response = await fetch(`/api/upload/progress/${taskId}`)
      
      if (!response.ok) {
        throw new Error('获取进度失败')
      }
      
      const data = await response.json()
      
      // 更新进度
      processingStage.value = data.stage || 'processing'
      processingProgress.loaded = data.processed || 0
      processingProgress.total = data.total || 100
      processingProgress.percent = data.percent || 
        Math.round((data.processed / data.total) * 100) || 0
      
      statusMessage.value = data.message
      
      // 检查状态
      if (data.status === 'completed') {
        processing.value = false
        uploadResult.value = {
          count: data.processed,
          ...data.result
        }
      } else if (data.status === 'failed') {
        handleError(data.message || '处理失败')
      } else {
        // 继续轮询
        retryCount = 0
        setTimeout(poll, pollInterval)
      }
    } catch (error) {
      retryCount++
      
      if (retryCount >= maxRetries) {
        handleError('轮询超时')
      } else {
        setTimeout(poll, pollInterval)
      }
    }
  }
  
  poll()
}

/**
 * 处理错误
 */
const handleError = (message) => {
  uploading.value = false
  processing.value = false
  uploadError.value = message
  clearInterval(speedTimer)
}

/**
 * 取消上传
 */
const cancelUpload = () => {
  if (window.currentXHR) {
    window.currentXHR.abort()
  }
  uploading.value = false
  uploadProgress.percent = 0
  clearInterval(speedTimer)
}
```

## 📊 方案对比

| 方案 | 适用场景 | 优点 | 缺点 |
|:---|:---|:---|:---|
| XHR 进度事件 | 简单文件上传 | 实现简单，进度精确 | 只能监控上传进度 |
| Fetch + Stream | 需要监控下载进度 | 现代 API | 无法监控上传进度 |
| WebSocket | 实时性要求高 | 实时推送，双向通信 | 需要服务器支持 |
| 轮询查询 | 后端处理时间长 | 通用性强，易于实现 | 有延迟，增加服务器负载 |

## 💡 最佳实践建议

1. **小文件**（<10MB）：使用 XHR 进度事件就够了
2. **大文件**（>100MB）：建议分片上传 + WebSocket 进度推送
3. **处理时间长**：上传完成后用轮询或 WebSocket 监控处理进度
4. **用户体验**：显示上传速度、预估剩余时间、处理阶段
5. **错误处理**：提供重试机制和取消功能

选择哪种方案取决于你的后端支持和业务需求。通常组合使用：**XHR 监控上传进度 + 轮询/WebSocket 监控处理进度** 是最佳实践。
