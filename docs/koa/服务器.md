我来帮你搭建一个基于 Node.js + Koa 的服务器。

## 1. 初始化项目

```bash
mkdir koa-server
cd koa-server
npm init -y
```

## 2. 安装依赖

```bash
npm install koa koa-router koa-bodyparser koa-static koa-compose
npm install -D nodemon
```

## 3. 创建服务器文件

### `app.js` - 主入口文件

```javascript
const Koa = require('koa');
const Router = require('koa-router');
const bodyParser = require('koa-bodyparser');
const serve = require('koa-static');
const path = require('path');

// 创建 Koa 应用
const app = new Koa();
const router = new Router();

// 中间件
// 1. 错误处理中间件
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      code: ctx.status,
      message: err.message,
      data: null
    };
    // 触发应用级别的错误事件
    ctx.app.emit('error', err, ctx);
  }
});

// 2. 日志中间件
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  console.log(`${ctx.method} ${ctx.url} - ${ms}ms`);
});

// 3. 解析请求体
app.use(bodyParser());

// 4. 静态文件服务
app.use(serve(path.join(__dirname, 'public')));

// 路由配置
// 测试路由
router.get('/', async (ctx) => {
  ctx.body = {
    code: 200,
    message: 'Koa 服务器运行成功！',
    data: {
      timestamp: new Date().toISOString()
    }
  };
});

// 获取用户信息
router.get('/api/user/:id', async (ctx) => {
  const { id } = ctx.params;
  ctx.body = {
    code: 200,
    message: '获取用户成功',
    data: {
      id,
      name: `User ${id}`,
      email: `user${id}@example.com`
    }
  };
});

// 创建用户
router.post('/api/user', async (ctx) => {
  const { name, email } = ctx.request.body;
  
  if (!name || !email) {
    ctx.status = 400;
    ctx.body = {
      code: 400,
      message: '姓名和邮箱不能为空',
      data: null
    };
    return;
  }
  
  ctx.body = {
    code: 201,
    message: '创建用户成功',
    data: {
      id: Date.now(),
      name,
      email,
      createdAt: new Date().toISOString()
    }
  };
});

// 更新用户
router.put('/api/user/:id', async (ctx) => {
  const { id } = ctx.params;
  const { name, email } = ctx.request.body;
  
  ctx.body = {
    code: 200,
    message: '更新用户成功',
    data: {
      id,
      name: name || `User ${id}`,
      email: email || `user${id}@example.com`,
      updatedAt: new Date().toISOString()
    }
  };
});

// 删除用户
router.delete('/api/user/:id', async (ctx) => {
  const { id } = ctx.params;
  
  ctx.body = {
    code: 200,
    message: '删除用户成功',
    data: {
      id,
      deletedAt: new Date().toISOString()
    }
  };
});

// 查询参数示例
router.get('/api/search', async (ctx) => {
  const { keyword, page = 1, limit = 10 } = ctx.query;
  
  ctx.body = {
    code: 200,
    message: '查询成功',
    data: {
      keyword: keyword || '无',
      page: parseInt(page),
      limit: parseInt(limit),
      results: [
        { id: 1, title: '示例结果1' },
        { id: 2, title: '示例结果2' }
      ]
    }
  };
});

// 应用路由
app.use(router.routes());
app.use(router.allowedMethods());

// 错误事件监听
app.on('error', (err, ctx) => {
  console.error('服务器错误:', err);
});

// 启动服务器
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`服务器已启动: http://localhost:${PORT}`);
  console.log(`环境: ${process.env.NODE_ENV || 'development'}`);
});
```

## 5. 创建工具函数

### `utils/response.js` - 统一响应格式

```javascript
class ResponseUtil {
  static success(ctx, data = null, message = '成功', code = 200) {
    ctx.status = code;
    ctx.body = {
      code,
      message,
      data,
      timestamp: new Date().toISOString()
    };
  }
  
  static error(ctx, message = '失败', code = 500, data = null) {
    ctx.status = code;
    ctx.body = {
      code,
      message,
      data,
      timestamp: new Date().toISOString()
    };
  }
  
  static created(ctx, data = null, message = '创建成功') {
    this.success(ctx, data, message, 201);
  }
  
  static badRequest(ctx, message = '请求参数错误') {
    this.error(ctx, message, 400);
  }
  
  static unauthorized(ctx, message = '未授权') {
    this.error(ctx, message, 401);
  }
  
  static forbidden(ctx, message = '禁止访问') {
    this.error(ctx, message, 403);
  }
  
  static notFound(ctx, message = '资源不存在') {
    this.error(ctx, message, 404);
  }
}

module.exports = ResponseUtil;
```

## 6. 添加路由模块化

### `routes/users.js` - 用户路由

```javascript
const Router = require('koa-router');
const ResponseUtil = require('../utils/response');

const router = new Router({
  prefix: '/api/users'
});

// 模拟数据库
let users = [
  { id: 1, name: '张三', email: 'zhangsan@example.com' },
  { id: 2, name: '李四', email: 'lisi@example.com' }
];

// 获取所有用户
router.get('/', async (ctx) => {
  ResponseUtil.success(ctx, users, '获取用户列表成功');
});

// 获取单个用户
router.get('/:id', async (ctx) => {
  const { id } = ctx.params;
  const user = users.find(u => u.id === parseInt(id));
  
  if (!user) {
    ResponseUtil.notFound(ctx, '用户不存在');
    return;
  }
  
  ResponseUtil.success(ctx, user, '获取用户成功');
});

// 创建用户
router.post('/', async (ctx) => {
  const { name, email } = ctx.request.body;
  
  if (!name || !email) {
    ResponseUtil.badRequest(ctx, '姓名和邮箱不能为空');
    return;
  }
  
  const newUser = {
    id: users.length + 1,
    name,
    email,
    createdAt: new Date().toISOString()
  };
  
  users.push(newUser);
  ResponseUtil.created(ctx, newUser, '创建用户成功');
});

// 更新用户
router.put('/:id', async (ctx) => {
  const { id } = ctx.params;
  const { name, email } = ctx.request.body;
  
  const index = users.findIndex(u => u.id === parseInt(id));
  
  if (index === -1) {
    ResponseUtil.notFound(ctx, '用户不存在');
    return;
  }
  
  users[index] = {
    ...users[index],
    name: name || users[index].name,
    email: email || users[index].email,
    updatedAt: new Date().toISOString()
  };
  
  ResponseUtil.success(ctx, users[index], '更新用户成功');
});

// 删除用户
router.delete('/:id', async (ctx) => {
  const { id } = ctx.params;
  const index = users.findIndex(u => u.id === parseInt(id));
  
  if (index === -1) {
    ResponseUtil.notFound(ctx, '用户不存在');
    return;
  }
  
  users.splice(index, 1);
  ResponseUtil.success(ctx, null, '删除用户成功');
});

module.exports = router;
```

### `routes/index.js` - 路由汇总

```javascript
const fs = require('fs');
const path = require('path');

module.exports = (app) => {
  // 自动加载所有路由文件
  const routesPath = __dirname;
  const files = fs.readdirSync(routesPath);
  
  files.forEach(file => {
    if (file !== 'index.js' && file.endsWith('.js')) {
      const route = require(path.join(routesPath, file));
      app.use(route.routes());
      app.use(route.allowedMethods());
    }
  });
};
```

## 7. 更新主文件 `app.js`

```javascript
const Koa = require('koa');
const bodyParser = require('koa-bodyparser');
const serve = require('koa-static');
const path = require('path');
const config = require('./config');
const ResponseUtil = require('./utils/response');

const app = new Koa();

// 中间件
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    if (err.status === 404) {
      ResponseUtil.notFound(ctx, '请求的资源不存在');
    } else {
      ResponseUtil.error(ctx, err.message, err.status || 500);
    }
    ctx.app.emit('error', err, ctx);
  }
});

app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  console.log(`${ctx.method} ${ctx.url} - ${ms}ms - ${ctx.status}`);
});

app.use(bodyParser());
app.use(serve(path.join(__dirname, 'public')));

// 加载路由
require('./routes')(app);

app.on('error', (err, ctx) => {
  console.error('服务器错误:', err.message);
});

const PORT = config.server.port;
app.listen(PORT, () => {
  console.log(`🚀 服务器已启动: http://${config.server.host}:${PORT}`);
  console.log(`📝 环境: ${process.env.NODE_ENV || 'development'}`);
});
```
