我来帮你增加 HTML 页面模板渲染功能。我会添加多种模板引擎的支持。

## 1. 安装模板引擎依赖

```bash
# 选择其中一种或多种
npm install koa-views ejs        # EJS 模板引擎
# 或
npm install koa-views pug        # Pug 模板引擎
# 或
npm install koa-views nunjucks   # Nunjucks 模板引擎
# 或安装多个
npm install koa-views ejs pug nunjucks
```

## 2. 更新主配置文件 `app.js`

```javascript
const Koa = require('koa');
const Router = require('koa-router');
const bodyParser = require('koa-bodyparser');
const serve = require('koa-static');
const views = require('koa-views');
const path = require('path');
const config = require('./config');

const app = new Koa();
const router = new Router();

// 配置模板引擎（使用 EJS）
app.use(views(path.join(__dirname, 'views'), {
  extension: 'ejs',
  map: { html: 'ejs' }
}));

// 中间件
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      code: ctx.status,
      message: err.message,
      data: null
    };
    ctx.app.emit('error', err, ctx);
  }
});

// 日志中间件
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  console.log(`${ctx.method} ${ctx.url} - ${ms}ms`);
});

app.use(bodyParser());
app.use(serve(path.join(__dirname, 'public')));

// ========== HTML 页面路由 ==========

// 首页
router.get('/', async (ctx) => {
  await ctx.render('index', {
    title: 'Koa 服务器',
    message: '欢迎使用 Koa + EJS 模板引擎',
    user: { name: '访客', role: '普通用户' },
    year: new Date().getFullYear()
  });
});

// 用户列表页面
router.get('/users', async (ctx) => {
  const users = [
    { id: 1, name: '张三', email: 'zhangsan@example.com', age: 25, role: '管理员' },
    { id: 2, name: '李四', email: 'lisi@example.com', age: 30, role: '普通用户' },
    { id: 3, name: '王五', email: 'wangwu@example.com', age: 28, role: '编辑' },
    { id: 4, name: '赵六', email: 'zhaoliu@example.com', age: 35, role: '普通用户' }
  ];
  
  await ctx.render('users', {
    title: '用户列表',
    users: users,
    total: users.length
  });
});

// 用户详情页面
router.get('/user/:id', async (ctx) => {
  const { id } = ctx.params;
  
  // 模拟数据
  const users = {
    1: { id: 1, name: '张三', email: 'zhangsan@example.com', age: 25, role: '管理员', bio: '热爱编程的全栈开发者' },
    2: { id: 2, name: '李四', email: 'lisi@example.com', age: 30, role: '普通用户', bio: '喜欢阅读和旅行' }
  };
  
  const user = users[id];
  
  if (!user) {
    ctx.status = 404;
    await ctx.render('404', { title: '页面未找到' });
    return;
  }
  
  await ctx.render('user-detail', {
    title: `${user.name}的详情`,
    user: user
  });
});

// 关于页面
router.get('/about', async (ctx) => {
  await ctx.render('about', {
    title: '关于我们',
    team: [
      { name: '张三', position: '技术负责人' },
      { name: '李四', position: '产品经理' },
      { name: '王五', position: 'UI设计师' }
    ]
  });
});

// 联系我们页面
router.get('/contact', async (ctx) => {
  await ctx.render('contact', {
    title: '联系我们',
    contact: {
      address: '北京市朝阳区xxx路123号',
      phone: '010-12345678',
      email: 'contact@example.com'
    }
  });
});

// 带查询参数的页面
router.get('/search-page', async (ctx) => {
  const { keyword = '', page = 1 } = ctx.query;
  
  // 模拟搜索结果
  const results = keyword ? [
    { title: `搜索结果1 - ${keyword}`, desc: '这是第一个搜索结果' },
    { title: `搜索结果2 - ${keyword}`, desc: '这是第二个搜索结果' }
  ] : [];
  
  await ctx.render('search', {
    title: '搜索页面',
    keyword: keyword,
    page: parseInt(page),
    results: results,
    total: results.length
  });
});

// ========== API 路由（保持不变） ==========
router.get('/api/hello', async (ctx) => {
  ctx.body = {
    code: 200,
    message: 'API 接口正常',
    data: { time: new Date().toISOString() }
  };
});

// 应用路由
app.use(router.routes());
app.use(router.allowedMethods());

// 错误事件监听
app.on('error', (err, ctx) => {
  console.error('服务器错误:', err);
});

// 启动服务器
const PORT = config.server.port;
app.listen(PORT, () => {
  console.log(`🚀 服务器已启动: http://${config.server.host}:${PORT}`);
  console.log(`📝 环境: ${process.env.NODE_ENV || 'development'}`);
  console.log(`🎨 模板引擎: EJS`);
});
```

## 4. 添加联系人 API（在 routes 中添加）

在 `app.js` 中添加处理联系表单的 API：

```javascript
// 处理联系表单提交
router.post('/api/contact', async (ctx) => {
  const { name, email, message } = ctx.request.body;
  
  if (!name || !email || !message) {
    ctx.body = {
      code: 400,
      message: '所有字段都是必填的'
    };
    return;
  }
  
  // 这里可以添加发送邮件或保存到数据库的逻辑
  console.log('收到联系消息:', { name, email, message });
  
  // 返回成功响应
  ctx.body = {
    code: 200,
    message: '消息发送成功，我们会尽快联系您！',
    data: { name, email }
  };
});
```
