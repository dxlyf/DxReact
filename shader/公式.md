# 图形学核心公式大全

## 一、基础几何公式

### 1. 向量运算
```glsl
// 向量长度
float length(vec3 v) = sqrt(v.x*v.x + v.y*v.y + v.z*v.z);

// 向量点积（内积）
float dot(vec3 a, vec3 b) = a.x*b.x + a.y*b.y + a.z*b.z;

// 向量叉积（外积）
vec3 cross(vec3 a, vec3 b) = vec3(
    a.y*b.z - a.z*b.y,
    a.z*b.x - a.x*b.z,
    a.x*b.y - a.y*b.x
);

// 向量投影
vec3 project(vec3 a, vec3 b) = (dot(a, b) / dot(b, b)) * b;
```

### 2. 距离公式
```glsl
// 点到点距离
float distancePointPoint(vec3 a, vec3 b) = length(a - b);

// 点到直线距离
float distancePointLine(vec3 p, vec3 linePoint, vec3 lineDir) {
    vec3 v = p - linePoint;
    vec3 w = cross(v, lineDir);
    return length(w) / length(lineDir);
}

// 点到平面距离
float distancePointPlane(vec3 p, vec4 plane) {
    // plane = (a, b, c, d) for ax + by + cz + d = 0
    return dot(plane.xyz, p) + plane.w;
}
```

## 二、2D图形公式

### 1. 基本2D图形
```glsl
// 圆
float circle(vec2 p, vec2 center, float radius) {
    return length(p - center) - radius;
}

// 矩形
float rectangle(vec2 p, vec2 center, vec2 size) {
    vec2 d = abs(p - center) - size * 0.5;
    return length(max(d, 0.0)) + min(max(d.x, d.y), 0.0);
}

// 椭圆
float ellipse(vec2 p, vec2 center, vec2 radii) {
    vec2 d = (p - center) / radii;
    return length(d) - 1.0;
}

// 正多边形（n边形）
float ngon(vec2 p, vec2 center, float radius, int n) {
    float angle = 2.0 * PI / float(n);
    vec2 d = p - center;
    float a = atan(d.y, d.x);
    a = mod(a, angle) - 0.5 * angle;
    return length(d) * cos(a) - radius;
}

// 星形
float star(vec2 p, vec2 center, float r1, float r2, int n) {
    vec2 d = p - center;
    float angle = 2.0 * PI / float(n);
    float a = atan(d.y, d.x);
    float segment = floor(a / angle + 0.5);
    float a1 = angle * segment;
    float a2 = angle * (segment + 0.5);
    vec2 p1 = center + r1 * vec2(cos(a1), sin(a1));
    vec2 p2 = center + r2 * vec2(cos(a2), sin(a2));
    return min(distance(p, p1), distance(p, p2));
}
```

### 3. 2D曲线公式
```glsl
// 贝塞尔曲线（二次）
vec2 quadraticBezier(vec2 p0, vec2 p1, vec2 p2, float t) {
    float u = 1.0 - t;
    return u*u*p0 + 2.0*u*t*p1 + t*t*p2;
}

// 贝塞尔曲线（三次）
vec2 cubicBezier(vec2 p0, vec2 p1, vec2 p2, vec2 p3, float t) {
    float u = 1.0 - t;
    return u*u*u*p0 + 3.0*u*u*t*p1 + 3.0*u*t*t*p2 + t*t*t*p3;
}

// 玫瑰线（Rose Curve）
vec2 roseCurve(float t, int n, float d, float a) {
    float k = float(n) / float(d);
    float r = a * cos(k * t);
    return vec2(r * cos(t), r * sin(t));
}
```

## 三、3D图形公式（SDF - 符号距离函数）

### 1. 基本3D图形
```glsl
// 球体
float sphere(vec3 p, float radius) {
    return length(p) - radius;
}

// 立方体
float box(vec3 p, vec3 size) {
    vec3 d = abs(p) - size * 0.5;
    return length(max(d, 0.0)) + min(max(d.x, max(d.y, d.z)), 0.0);
}

// 圆柱体
float cylinder(vec3 p, float radius, float height) {
    vec2 d = abs(vec2(length(p.xz), p.y)) - vec2(radius, height * 0.5);
    return min(max(d.x, d.y), 0.0) + length(max(d, 0.0));
}

// 圆锥
float cone(vec3 p, float radius, float height) {
    vec2 q = vec2(length(p.xz), p.y);
    vec2 tip = q - vec2(0, height);
    vec2 mantleDir = normalize(vec2(height, radius));
    float mantle = dot(tip, mantleDir);
    float d = max(mantle, -p.y);
    float projected = dot(tip, vec2(mantleDir.y, -mantleDir.x));
    
    // 侧面
    if(p.y > height && projected < 0.0) {
        d = max(d, tip.length());
    }
    
    // 底面
    if(q.x < radius && p.y < 0.0) {
        d = max(d, -p.y);
    }
    
    return d;
}

// 圆环
float torus(vec3 p, float majorRadius, float minorRadius) {
    vec2 q = vec2(length(p.xz) - majorRadius, p.y);
    return length(q) - minorRadius;
}
```

```glsl
// 3D心形（经典公式）
float heart3D(vec3 p, float size) {
    p /= size;
    p.y -= 0.25;
    float x2 = p.x * p.x;
    float y2 = p.y * p.y;
    float z2 = p.z * p.z;
    
    float a = x2 + 2.25 * y2 + z2 - 1.0;
    float b = x2 * z2 * 0.25 + y2 * z2;

}

float simpleHeart3D(vec3 p) {

}
```

### 3. 3D复杂图形
```glsl
// 胶囊体
float capsule(vec3 p, vec3 a, vec3 b, float radius) {
    vec3 ab = b - a;
    vec3 ap = p - a;
    float t = dot(ap, ab) / dot(ab, ab);
    t = clamp(t, 0.0, 1.0);
    vec3 c = a + t * ab;
    return length(p - c) - radius;
}

// 椭球体
float ellipsoid(vec3 p, vec3 radii) {
    vec3 d = p / radii;
    float k0 = length(d);
    float k1 = length(p / (radii * radii));
    return k0 * (k0 - 1.0) / k1;
}

// 四面体
float tetrahedron(vec3 p, float size) {
    vec3 a1 = normalize(vec3(1, 1, 1));
    vec3 a2 = normalize(vec3(-1, -1, 1));
    vec3 a3 = normalize(vec3(-1, 1, -1));
    vec3 a4 = normalize(vec3(1, -1, -1));
    
    float d1 = dot(p, a1) - size;
    float d2 = dot(p, a2) - size;
    float d3 = dot(p, a3) - size;
    float d4 = dot(p, a4) - size;
    
    return max(max(d1, d2), max(d3, d4));
}

// 八面体
float octahedron(vec3 p, float size) {
    p = abs(p);
    float m = p.x + p.y + p.z - size;
    
    vec3 q;
    if(3.0 * p.x < m) q = p;
    else if(3.0 * p.y < m) q = p.yzx;
    else if(3.0 * p.z < m) q = p.zxy;
    else return m * 0.57735027;
    
    float k = clamp(0.5 * (q.z - q.y + size), 0.0, size);
    return length(vec3(q.x, q.y - size + k, q.z - k));
}

// 二十面体
float icosahedron(vec3 p, float size) {
    p = abs(p);
    float m = dot(p, vec3(0.57735027));
    
    const vec3 n = vec3(0.934172, 0.356822, 0.0);
    float d = dot(p, n);
    
    p = abs(p * 1.7320508 - m);
    if(d > 0.0) p = p.yzx;
    
    p = abs(p);
    m = max(p.x + p.y, max(p.y + p.z, p.z + p.x));
    
    return (m - size) * 0.57735027;
}
```

### 4. 3D曲面公式
```glsl
// 莫比乌斯带
float mobiusStrip(vec3 p, float radius, float width) {
    float u = atan(p.z, p.x);
    float v = p.y;
    
    vec3 q = vec3(
        (radius + v * cos(0.5 * u)) * cos(u),
        (radius + v * cos(0.5 * u)) * sin(u),
        v * sin(0.5 * u)
    );
    
    return length(p - q) - width;
}

// 克莱因瓶
float kleinBottle(vec3 p, float size) {
    p /= size;
    float u = atan(p.z, p.x);
    float v = p.y;
    
    float x = (1.0 + 0.5 * cos(v)) * cos(u);
    float y = (1.0 + 0.5 * cos(v)) * sin(u);
    float z = 0.5 * sin(v);
    
    if(u < PI) {
        x = 0.5 * cos(u) * (1.0 + sin(v));
        y = 0.5 * sin(u) * (1.0 + sin(v));
        z = 0.5 * cos(v) + 0.5;
    }
    
    return length(p - vec3(x, y, z)) - 0.1;
}

// 螺旋
float helix(vec3 p, float radius, float pitch) {
    float t = atan(p.z, p.x);
    vec3 q = vec3(
        radius * cos(t),
        pitch * t / (2.0 * PI),
        radius * sin(t)
    );
    return length(p - q) - 0.1;
}
```

```glsl
vec3 toSpherical(vec3 cartesian) {
    float r = length(cartesian);
    float theta = atan(cartesian.z, cartesian.x);

    return vec3(r, theta, phi);

vec3 fromSpherical(vec3 spherical) {
    float r = spherical.x;
    float theta = spherical.y;
    float phi = spherical.z;

        r * cos(phi),
        r * sin(phi) * sin(theta)
    );
}
```

## 六、分形公式

### 1. 2D分形
```glsl
// 曼德勃罗集
float mandelbrot(vec2 c, int iterations) {
    vec2 z = vec2(0.0);
    for(int i = 0; i < iterations; i++) {
        z = vec2(z.x*z.x - z.y*z.y, 2.0*z.x*z.y) + c;
        if(length(z) > 2.0) return float(i) / float(iterations);
    }
    return 1.0;
}

// 茱莉亚集
float julia(vec2 z, vec2 c, int iterations) {
    for(int i = 0; i < iterations; i++) {
        z = vec2(z.x*z.x - z.y*z.y, 2.0*z.x*z.y) + c;
        if(length(z) > 2.0) return float(i) / float(iterations);
    }
    return 1.0;
}

// 谢尔宾斯基三角形
float sierpinskiTriangle(vec2 p) {
    p = abs(p - 0.5);
    if(p.x + p.y > 1.0) p = 1.0 - p;
    
    float scale = 1.0;
    for(int i = 0; i < 10; i++) {
        p = 2.0 * p - vec2(1.0);
        scale *= 2.0;
        if(p.x + p.y < 0.0) {
            p = -p;
        } else if(p.x - p.y > 0.0) {
            p = p.yx;
        }
    }
    return abs(p.x - p.y) * scale;
}
```

### 2. 3D分形
```glsl
// 曼德尔球
float mandelbulb(vec3 p, int iterations, float power) {
    vec3 z = p;
    float dr = 1.0;
    float r = 0.0;
    
    for(int i = 0; i < iterations; i++) {
        r = length(z);
        if(r > 2.0) break;
        
        // 转换为球坐标
        float theta = atan(z.y, z.x);
        float phi = acos(z.z / r);
        
        // 应用幂运算
        float zr = pow(r, power);
        theta *= power;
        phi *= power;
        
        // 转换回直角坐标
        z = zr * vec3(sin(phi) * cos(theta), sin(phi) * sin(theta), cos(phi));
        z += p;
        
        dr = pow(r, power - 1.0) * power * dr + 1.0;
    }
    
    return 0.5 * log(r) * r / dr;
}

// 门格海绵
float mengerSponge(vec3 p) {
    p = abs(p - 0.5);
    if(max(p.x, max(p.y, p.z)) > 0.5) return 0.5;
    
    float scale = 1.0;
    for(int i = 0; i < 5; i++) {
        p *= 3.0;
        scale *= 3.0;
        vec3 a = mod(p, 3.0);
        vec3 b = floor(p / 3.0);
        
        if(a.x < 1.0 && a.y < 1.0 && a.z < 1.0) continue;
        if(a.x < 1.0 && a.y < 1.0) continue;
        if(a.y < 1.0 && a.z < 1.0) continue;
        if(a.z < 1.0 && a.x < 1.0) continue;
        
        return 0.5 / scale;
    }
    
    return 0.5 / scale;
}
```

### 2. 噪声函数
```glsl
// 值噪声
float valueNoise(vec2 p) {
    vec2 i = floor(p);
    vec2 f = fract(p);
    
    float a = hash(i);
    float b = hash(i + vec2(1.0, 0.0));
    float c = hash(i + vec2(0.0, 1.0));
    float d = hash(i + vec2(1.0, 1.0));
    
    vec2 u = f * f * (3.0 - 2.0 * f);
    return mix(a, b, u.x) + (c - a) * u.y * (1.0 - u.x) + (d - b) * u.x * u.y;
}

// 梯度噪声（Perlin噪声）
float gradientNoise(vec2 p) {
    vec2 i = floor(p);
    vec2 f = fract(p);
    
    vec2 u = f * f * (3.0 - 2.0 * f);
    
    float a = dot(hash22(i + vec2(0.0, 0.0)), f - vec2(0.0, 0.0));
    float b = dot(hash22(i + vec2(1.0, 0.0)), f - vec2(1.0, 0.0));
    float c = dot(hash22(i + vec2(0.0, 1.0)), f - vec2(0.0, 1.0));
    float d = dot(hash22(i + vec2(1.0, 1.0)), f - vec2(1.0, 1.0));
    
    return mix(mix(a, b, u.x), mix(c, d, u.x), u.y);
}
```

```glsl
mat4 lookAt(vec3 eye, vec3 target, vec3 up) {
    vec3 z = normalize(eye - target);

    vec3 y = cross(z, x);

mat4 perspective(float fov, float aspect, float near, float far) {
    float tanHalfFov = tan(fov * 0.5);

        vec4(0.0, 1.0 / tanHalfFov, 0.0, 0.0),
        vec4(0.0, 0.0, -(far + near) / (far - near), -1.0),
        vec4(0.0, 0.0, -2.0 * far * near / (far - near), 0.0)
    );
}
```

## 九、动画与特效公式

### 1. 缓动函数

```glsl
float easeIn(float t, float power) {

    return sin(13.0 * t * PI/2.0) * pow(2.0, 10.0 * (t - 1.0));
}
```

```glsl
float mapRange(float value, float inMin, float inMax, float outMin, float outMax) {

vec3 linearToSRGB(vec3 linear) {
    return mix(
        linear * 12.92,
        1.055 * pow(linear, vec3(1.0/2.4)) - 0.055,
        step(0.0031308, linear)
    );
}

vec3 sRGBToLinear(vec3 srgb) {
    return mix(
        srgb / 12.92,
        pow((srgb + 0.055) / 1.055, vec3(2.4)),
        step(0.04045, srgb)
    );
}
```

## 使用示例

```glsl
// 组合使用示例：创建一个跳动的心形
void mainImage(out vec4 fragColor, in vec2 fragCoord) {
    vec2 uv = (fragCoord - 0.5 * iResolution.xy) / iResolution.y;
    float time = iTime;
    
    // 缩放和移动
    uv *= 1.5;
    uv.y += 0.3;
    
    // 心跳动画
    float beat = sin(time * 5.0) * exp(-time * 0.5);
    uv *= 0.9 + 0.1 * beat;
    
    // 计算心形
    float heart = heart2D(uv);
    
    // 颜色
    vec3 color = vec3(0.0);
    if(heart < 0.02) {
        // 使用渐变色
        float dist = abs(heart) * 50.0;
        color = mix(
            vec3(1.0, 0.2, 0.4),  // 中心色
            vec3(0.9, 0.1, 0.3),  // 边缘色
            smoothstep(0.0, 1.0, dist)
        );
        
        // 添加高光
        vec2 normal = vec2(
            heart2D(uv + vec2(0.001, 0.0)) - heart2D(uv - vec2(0.001, 0.0)),
            heart2D(uv + vec2(0.0, 0.001)) - heart2D(uv - vec2(0.0, 0.001))
        );
        normal = normalize(normal);
        
        // 模拟光照
        vec2 lightDir = normalize(vec2(0.5, 1.0));
        float diffuse = max(dot(normal, lightDir), 0.0);
        color *= 0.5 + 0.5 * diffuse;
    }
    
    fragColor = vec4(color, 1.0);
}
```

这个大全包含了图形学中最常用的公式，从基础几何到复杂分形，从2D图形到3D曲面。你可以根据需要组合使用这些公式来创建各种视觉效果！
